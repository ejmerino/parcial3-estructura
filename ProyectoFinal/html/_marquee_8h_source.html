93char* Marquee::marquee(int position) {
94 int counter = 0;
95 char* display = (char*)malloc(sizeof(char) * (lenght + 1));
96 char* travel = (char*)malloc(sizeof(char) * 256);
97 char* temp = (char*)malloc(sizeof(char) * (strlen(content) + 1));
98 string_op.strcpy(travel, content);
99 cursorHide();
100 while (counter < position) {
101 temp[counter] = travel[0];
102 travel++;
103 counter = counter + 1;
104 }
105 temp[counter + 1] = '\0';
106 string_op.strcat(travel, " ");
107 string_op.strcat(travel, temp);
108 string_op.strncpy(display, travel, lenght);
109 display[lenght] = '\0';
110 return display;
111}
