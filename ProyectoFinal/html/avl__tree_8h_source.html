108inline AvlNode* AvlTree::insertAvl(AvlNode* current, std::string artista, std::string cancion)
109{
110 if (current == nullptr)
111 {
112 size++;
113 return new AvlNode(cancion, artista);
114 }
115
116 if (cancion < current->cancion())
117 {
118 current->set_left(insertAvl(current->left(), artista, cancion));
119 }
120 else if (cancion >= current->cancion())
121 {
122 current->set_right(insertAvl(current->right(), artista, cancion));
123 }
124
125 current->set_height(max_height(find_height(current->left()), find_height(current->right())) + 1);
126
127 //* Obtiene el nuevo factor de equilibrio
128 int bf = balance_factor(current);
129
130 //* Rotacion simple a la derecha
131 if (bf > 1 && cancion < current->left()->cancion())
132 {
133 return rotar_der(current);
134 }
135
136 //* Rotacion simple a la izquierda
137 if (bf < -1 && cancion > current->right()->cancion())
138 {
139 return rotar_izq(current);
140 }
141
142 //* Rotacion izquierda derecha
143 if (bf > 1 && cancion > current->left()->cancion())
144 {
145 current->set_left(rotar_izq(current->left()));
146 return rotar_der(current);
147 }
148
149 //* Rotacion derecha izquierda
150 if (bf < -1 && cancion < current->right()->cancion())
151 {
152 current->set_right(rotar_der(current->right()));
153 return rotar_izq(current);
154 }
155 return current;
156}

158inline AvlNode* AvlTree::borrarAvl(AvlNode* current, std::string word)
159{
160 if (current == nullptr)
161 {
162 return current;
163 }
164 if (word < current->cancion())
165 {
166 current->set_left(borrarAvl(current->left(), word));
167 }
168 else if (word > current->cancion())
169 {
170 current->set_right(borrarAvl(current->right(), word));
171 }
172 else
173 {
174 if ((current->left() == nullptr) || (current->right() == nullptr))
175 {
176 AvlNode* temp = nullptr;
177
178 if (temp == current->left())
179 {
180 temp = current->right();
181 }
182 else
183 {
184 temp = current->left();
185 }
186
187 //* Si no hay hijos, se elimina el nodo
188 if (temp == nullptr)
189 {
190 current = nullptr;
191 size--;
192 }
193 else
194 {
195 current = temp;
196 }
197 }
198 else
199 {
200 //* Si hay hijos, se busca el nodo m�s alto
201 AvlNode* temp = max_value_node(current->left());
202
203 //* Se asigna el valor del nodo m�s alto a la raiz
204 current->set_cancion(temp->cancion());
205 current->set_artista(temp->artista());
206
207 current->set_left(borrarAvl(current->left(), temp->cancion()));
208 }
209 }
210
211 if (current == nullptr)
212 {
213 return { current };
214 }
215
216 current->set_height(max_height(find_height(current->left()), find_height(current->right())) + 1);
217
218 //* Obtiene el nuevo factor de equilibrio
219 int bf = balance_factor(current);
220 //* Rotacion simple a la derecha
221 if (bf > 1 && balance_factor(current->left()) >= 0)
222 {
223 return rotar_der(current);
224 }
225
226 //* Rotacion simple a la izquierda
227 if (bf < -1 && balance_factor(current->right()) <= 0)
228 {
229 return rotar_izq(current);
230 }
231 //* Rotacion izquierda derecha
232 if (bf > 1 && balance_factor(current->left()) < 0)
233 {
234 current->set_left(rotar_izq(current->left()));
235 return rotar_der(current);
236 }
237 //* Rotacion derecha izquierda
238 if (bf < -1 && balance_factor(current->right()) > 0)
239 {
240 current->set_right(rotar_der(current->right()));
241 return rotar_izq(current);
242 }
243
244 return current;
245}

249inline void AvlTree::for_each_inorder(std::function<void(AvlNode*)> callback)
250{
251 //* Construct the following tree
252 //*
253 //* 5
254 //* / \
255 //* 3 7
256 //* / \ \
257 //* 1 4 8
258 //* / \
259 //* 6 9
260 //*
261 //* inorder traversal should be:
262 //* 1, 3, 4, 5, 6, 7, 8, 9
263 //*
264 if (root == nullptr)
265 {
266 return;
267 }
268 std::stack<AvlNode*> stack;
269 AvlNode* current = root;
270 while (!stack.empty() || current != nullptr)
271 {
272 if (current != nullptr)
273 {
274 stack.push(current);
275 current = current->left();
276 }
277 else
278 {
279 current = stack.top();
280 stack.pop();
281 callback(current);
282 current = current->right();
283 }
284 }
285}

293 std::stack<AvlNode*> stack;

298 stack.pop();
299 callback(current);
300 if (current->right() != nullptr)
301 {
302 stack.push(current->right());
303 }
304 if (current->left() != nullptr)
305 {
306 stack.push(current->left());
307 }

311inline void AvlTree::for_each_postorder(std::function<void(AvlNode*)> callback)
312{
313 if (root == nullptr)
314 {
315 return;
316 }
317 std::stack<AvlNode*> stack;
318 stack.push(root);
319 std::stack<AvlNode*> postorder;
320 while (!stack.empty())
321 {
322 AvlNode* current = stack.top();
323 stack.pop();
324 postorder.push(current);
325 if (current->left() != nullptr)
326 {
327 stack.push(current->left());
328 }
329 if (current->right() != nullptr)
330 {
331 stack.push(current->right());
332 }
333 }
334 while (!postorder.empty())
335 {
336 callback(postorder.top());
337 postorder.pop();
338 }
339}
